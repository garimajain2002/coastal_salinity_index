// Match Landsat5 with Landsat 8 to account for different sensor caliberation and radiometric properties between the two
// Function to match histogram of Landsat 5 images to Landsat 8
function histMatch(image, refImage) {
  var bands = ['BLUE_p30', 'GREEN_p30', 'RED_p30', 'NIR_p30', 'SWIR1_p30', 'SWIR2_p30', 'NDVI_p30', 'NDWI_p30', 'NDSI_p30']; 
  var matchedBands = bands.map(function(band) {
    var imageBand = image.select(band);
    var refBand = refImage.select(band);

    // Compute 5th and 95th percentiles for both images
    var imageStats = imageBand.reduceRegion({
      reducer: ee.Reducer.percentile([5, 95]),
      geometry: SamplingArea,
      scale: 30,
      maxPixels: 1e13
    });

    var refStats = refBand.reduceRegion({
      reducer: ee.Reducer.percentile([5, 95]),
      geometry: SamplingArea,
      scale: 30,
      maxPixels: 1e13
    });

    var imageMin = ee.Number(imageStats.get(band + '_p5'));
    var imageMax = ee.Number(imageStats.get(band + '_p95'));
    var refMin = ee.Number(refStats.get(band + '_p5'));
    var refMax = ee.Number(refStats.get(band + '_p95'));

    // Apply linear transformation
    var matchedBand = imageBand.subtract(imageMin)
      .multiply(refMax.subtract(refMin).divide(imageMax.subtract(imageMin)))
      .add(refMin)
      .rename(band + '_matched');

    return matchedBand;
  });

  return image.addBands(matchedBands, null, true);
}

// Define histogram plot function 
function plotHistogram(image, band, title) {
  var histogram = image.select(band).reduceRegion({
    reducer: ee.Reducer.histogram(255), // Bins set to 255
    geometry: SamplingArea,
    scale: 30,
    maxPixels: 1e13
  });

  // Extract histogram dictionary
  var histDict = ee.Dictionary(histogram.get(band));
  
  // Ensure we get actual values (avoid empty results)
  var histValues = ee.List(histDict.get('histogram'));
  var histBins = ee.List(histDict.get('bucketMeans'));

  // Create the chart if histValues is not empty
  var chart = ui.Chart.array.values(histValues, 0, histBins)
    .setOptions({
      title: title,
      hAxis: { title: 'Reflectance' },
      vAxis: { title: 'Frequency' },
      series: { 0: { color: 'blue' } }
    });

  print(chart);
}

    // Plot histogram for the reference Landsat 8 image
    plotHistogram(referenceImage, band, 'Reference Landsat 8 Histogram - ' + band);
  });
}

// Method 2: Mean-variance normalisation of Landsat 5 image bands with landsat 8 reference image 
// Function to compute mean-variance normalization for one band
function meanVarianceNormalization(image, referenceImage, band) {
  var refStats = referenceImage.select(band).reduceRegion({
    reducer: ee.Reducer.mean().combine({
      reducer2: ee.Reducer.stdDev(),
      sharedInputs: true
    }),
    geometry: SamplingArea,
    scale: 30,
    maxPixels: 1e13
  });

  var refMean = ee.Number(refStats.get(band + '_mean'));
  var refStdDev = ee.Number(refStats.get(band + '_stdDev'));

  var imgStats = image.select(band).reduceRegion({
    reducer: ee.Reducer.mean().combine({
      reducer2: ee.Reducer.stdDev(),
      sharedInputs: true
    }),
    geometry: SamplingArea,
    scale: 30,
    maxPixels: 1e13
  });

  var imgMean = ee.Number(imgStats.get(band + '_mean'));
  var imgStdDev = ee.Number(imgStats.get(band + '_stdDev'));

  // Apply normalization formula: (x - mean) / stdDev * refStdDev + refMean
  return image.select(band)
    .subtract(imgMean)
    .divide(imgStdDev)
    .multiply(refStdDev)
    .add(refMean)
    .rename(band);
}

// Apply mean-variance normalization to all bands
function normalizeBands(image, referenceImage, bands) {
  var normalizedBands = bands.map(function(band) {
    return meanVarianceNormalization(image, referenceImage, band);
  });

  // Combine all normalized bands into a single image
  return ee.Image.cat(normalizedBands).set('system:time_start', image.get('system:time_start'));
}
